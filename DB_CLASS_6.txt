#CREATING TABLES#
~~~~~~~~~~EMPLOYEE TABLE~~~~~~~~

	CREATE TABLE EMPLOYEE(
	ENO CHAR(4),FNAME VARCHAR(20),MINI CHAR(2),
	LNAME VARCHAR(20),DOB DATE,ADDR VARCHAR(20),
	SEX CHAR(2),SALARY NUMBER(5),SUPER_ENO CHAR(4),DNO CHAR(4),
	CONSTRAINT PK_EMP PRIMARY KEY(ENO),
	CONSTRAINT FK_EMP FOREIGN KEY(SUPER_ENO) REFERENCES EMPLOYEE(ENO));

~~~~~~~~~~DEPARTMENT TABLE~~~~~~~~~~~~~~

	CREATE TABLE DEPARTMENT(
	DNUM CHAR(4),DNAME VARCHAR(20),
	MGRENO CHAR(4),MGRSTRTDATE DATE,
	CONSTRAINT PK_DEPT PRIMARY KEY(DNUM),
	CONSTRAINT FK_DEPT FOREIGN KEY(MGRENO) REFERENCES EMPLOYEE(ENO));

#USING ALTER COMMAND TO REFERENCE DNUM FROM DEPARTMENT TABLE TO EMPLOYEE TABLE DNO#

	 ALTER TABLE EMPLOYEE ADD CONSTRAINT FK_EMP_DNO FOREIGN KEY(DNO) REFERENCES DEPARTMENT(DNUM);
	 
~~~~~~~~~~~DEPT_LOCATIONS TABLE~~~~~~~~~

	CREATE TABLE DEPT_LOCATIONS(
	DNUM CHAR(4),DLOC VARCHAR(20),
	CONSTRAINT PK_DLOC PRIMARY KEY(DNUM,DLOC),
	CONSTRAINT FK_DLOC FOREIGN KEY(DNUM) REFERENCES DEPARTMENT(DNUM));
 
 ~~~~~~~~~PROJECT TABLE~~~~~~~~
  
	CREATE TABLE PROJECT(
	PNO CHAR(4),PNAME VARCHAR(20),
	PLOC VARCHAR(20),DNUM CHAR(4),
	CONSTRAINT PK_PRJ PRIMARY KEY(PNO),
	CONSTRAINT FK_PRJ FOREIGN KEY(DNUM) REFERENCES DEPARTMENT(DNUM));

~~~~~~~~~~WORKS_ON~~~~~~~~~~~

	CREATE TABLE WORKS_ON(
	ENO CHAR(4), PNO CHAR(4), HOURS NUMBER(3),
	CONSTRAINT PK_WORKS PRIMARY KEY(ENO,PNO),
	CONSTRAINT FK_WORKS_EMP FOREIGN KEY(ENO) REFERENCES EMPLOYEE(ENO),
	CONSTRAINT FK_WORKS_PRJ FOREIGN KEY(PNO) REFERENCES PROJECT(PNO));
	
~~~~~~~~~~~DEPENDENT TABLE~~~~~~~~~~~~~~~~


	CREATE TABLE DEPENDENT(
	ENO CHAR(4),DEP_NAME VARCHAR(20),
	SEX CHAR(3),BDATE DATE,REL VARCHAR(10),
	CONSTRAINT PK_DEPEN  PRIMARY KEY(ENO,DEP_NAME,REL),
	CONSTRAINT FK_DEPEN_EMP FOREIGN KEY(ENO) REFERENCES EMPLOYEE(ENO));

#USING ALTER TABLE TO ADD CHECK CONSTRAINT#
	
	ALTER TABLE EMPLOYEE ADD CONSTRAINT CH_EMP CHECK (DOB<'01-JAN-2017');
	 
#RENAMING A COLUMN NAME USING ALTER TABLE#

	ALTER TABLE EMPLOYEE RENAME COLUMN ENO TO EMPNO;
	 
#RENAMING THE TABLE NAME#

	RENAME EMPLOYEE TO EMP;

#REMOVING FOREIGN KEY CONSTRAINT FROM A TABLE#

	ALTER TABLE WORKS_ON DROP CONSTRAINT FK_WORKS_EMP;
	
#ADDING A CONSTRAINT USING ALTER TABLE#

	ALTER TABLE WORKS_ON ADD CONSTRAINT FK_WORKS_EMP FOREIGN KEY(ENO) REFERENCES EMPLOYEE(EMPNO);
